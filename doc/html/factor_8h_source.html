1089  double distance(factor const& F2, Distance type = Distance::L2) const {
1090  assert(vars() == F2.vars());
1091  factor F(*this), Ftmp; // make a copy for manipulation
1092  double dist = -1.0; // local variables
1093  value Z;
1094  switch (type) {
1095  case Distance::L2: // L2, sum of squared errors
1096  F -= F2;
1097  F *= F;
1098  dist = F.sum();
1099  break;
1100  case Distance::L1: // L1, sum of absolute errors (=TV !!)
1101  F -= F2;
1102  dist = F.abs().sum();
1103  break;
1104  case Distance::LInf: // L-infinity, max absolute error
1105  F -= F2;
1106  dist = F.abs().max();
1107  break;
1108  case Distance::KL: // KL-divergence (relative entropy)
1109  Z = sum();
1110  F /= F2;
1111  F *= F2.sum() / Z;
1112  F.log();
1113  F *= *this;
1114  dist = F.sum() / Z;
1115  break;
1116  case Distance::HPM: // Hilbert's projective metric
1117  F /= F2;
1118  F.log();
1119  dist = F.max() - F.min(); // aka "dynamic range"
1120  break;
1121  case Distance::MAS: // "MAS" error value (not a metric)
1122  F.log();
1123  Ftmp = F2;
1124  F /= Ftmp.log();
1125  dist = std::max(F.max(), 1.0 / F.min()) - 1.0;
1126  break;
1127  case Distance::OptGap: // "Primal/Dual Gap"-like
1128  double mx1, mx2, gap1, gap2;
1129  mx1 = mx2 = gap1 = gap2 = 0.0;
1130  for (size_t i = 0; i < num_states(); ++i) {
1131  if (mx1 < F[i]) {
1132  gap2 = F2[i];
1133  mx1 = F[i];
1134  } else if (mx1 == F[i])
1135  gap2 = std::min(gap2, F2[i]);
1136  if (mx2 < F2[i]) {
1137  gap1 = F[i];
1138  mx2 = F2[i];
1139  } else if (mx2 == F2[i])
1140  gap1 = std::min(gap1, F[i]);
1141  }
1142  return (mx1 - gap1) + (mx2 - gap2);
1143  break;
1144  //case Distance::Hellinger: (!!)
1145  // F^=0.5; F-=F2^0.5; F*=F; dist=(0.5*F.sum())^0.5; // straightforward computation
1146  // F*=F2; F^=0.5; dist=(1-F.sum())^0.5; // alternate computation if F,F2 normalized
1147  // break;
1148  default:
1149  throw std::runtime_error("Invalid distance type");
1150  }
1151  return dist;
1152  };

1199  std::vector<factor> decomp_sum(std::vector<variable_set> vlist,
1200  factor::Decomp method) const {
1201  int nF = vlist.size();
1202  double mx, mn;
1203  std::vector<factor> Flist(nF);
1204 
1205  factor tmp, F = *this;
1206  switch (method) {
1207  case Decomp::L2: //L2
1208  double Cn, Cd;
1209  Cd = F.numel();
1210  Cn = F.sum(); // /Cd*(1-1.0/nF);
1211  for (int j = 0; j < nF; j++) {
1212  Flist[j] = F.marginal(vlist[j]);
1213  double D = Cd / Flist[j].numel();
1214  Flist[j] /= D;
1215  Flist[j] -= Cn / Cd * (1.0 - 1.0 / (nF - j));
1216  F -= Flist[j];
1217  Cn -= Flist[j].sum() * D;
1218  }
1219  break;
1220  case Decomp::L2_HPM: //L2+HPM
1221  Flist = decomp_sum(vlist, Decomp::L2);
1222  for (int j = 0; j < nF; j++)
1223  F -= Flist[j];
1224  mx = F.max();
1225  mn = F.min();
1226  for (int j = 0; j < nF; j++)
1227  Flist[j] += (mx + mn) / 2 / nF;
1228  break;
1229  case Decomp::L2_MAS: //L2+MAS
1230  Flist = decomp_sum(vlist, Decomp::L2);
1231  F = Flist[0];
1232  for (int j = 1; j < nF; j++)
1233  F += Flist[j];
1234  F /= *this;
1235  F.log();
1236  mx = F.max();
1237  mn = F.min();
1238  for (int j = 0; j < nF; j++)
1239  Flist[j] *= std::exp(-(mx + mn) / 2 / nF);
1240  break;
1241  }
1242  return Flist;
1243  }
