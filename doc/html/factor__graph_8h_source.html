92  findex add_factor(const merlin::factor& F) {
93  findex use = graphical_model::add_factor(F); // add the factor to the underlying collection
94  m_vindex.resize(nvar(), vindex(-1)); // then update variable nodes and edges (!!!)
95  if (F.nvar() == 1 && m_vindex[*F.vars().begin()] == vindex(-1))
96  m_vindex[*F.vars().begin()] = use;
97  else
98  for (variable_set::const_iterator v = F.vars().begin();
99  v != F.vars().end(); ++v) {
100  if (m_vindex[*v] == vindex(-1))
101  m_vindex[*v] = graphical_model::add_factor(merlin::factor(*v, 1.0));
102  add_edge(use, local_factor(*v)); // add edge to variable node
103  }
104  return use;
105  }

206  void check_factors() {
207  if (m_vindex.size() < nvar())
208  m_vindex.resize(nvar());
209  std::vector<bool> found(nvar(), false);
210  for (size_t i = 0; i < m_factors.size(); ++i) {
211  if (m_factors[i].nvar() == 1) {
212  size_t v = _vindex(*m_factors[i].vars().begin());
213  if (!found[v])
214  m_vindex[v] = i;
215  found[v] = true;
216  }
217  }
218  for (size_t v = 0; v < found.size(); ++v)
219  if (!found[v]) {
220  m_vindex[v]=
221  add_factor(merlin::factor(var(v), 1.0));
222  }
223  for (size_t i = 0; i < m_factors.size(); ++i) {
224  if (!is_var_node(i))
225  for (variable_set::const_iterator v = m_factors[i].vars().begin();
226  v != m_factors[i].vars().end(); ++v)
227  add_edge(i, local_factor(*v));
228  }

311  std::vector<edge_t> span_tree(TreeType tt, variable root) {
312  abstract_queue<findex> *Q;
313  switch (tt) {
314  case TreeType::WidthFirst:
315  Q = new abstract_queue_fifo<findex>();
316  break;
317  case TreeType::DepthFirst:
318  Q = new abstract_queue_lifo<findex>();
319  break;
320  case TreeType::MaxWeight:
321  default:
322  throw std::runtime_error("Not implemented");
323  break;
324  }
325  Q->push(local_factor(root));
326  std::vector<findex> used, fUsed;
327  used.resize(nvar(), findex(-1));
328  fUsed.resize(num_factors(), 0);
329  std::vector<edge_t> tree;
330  tree.reserve(2 * nvar());
331  while (!Q->empty()) {
332  findex next = Q->first();
333  Q->pop();
334  if (fUsed[next] != 0)
335  continue; // already taken this factor?
336  size_t nFound = 0, parent = 0;// if not look over variables and make sure
337  const variable_set& vs = get_factor(next).vars(); // at most one is already covered;
338  for (variable_set::const_iterator v = vs.begin(); v != vs.end(); ++v) {
339  if (used[_vindex(*v)] != vindex(-1)) {
340  parent = used[_vindex(*v)];
341  nFound++;
342  }
343  } // if one is covered that's our parent; more
344  if (nFound > 1)
345  continue;// means this factor is unavailable so keep looking
346  fUsed[next] = 1; // otherwise, mark factor, variables as covered
347  if (is_var_node(next))
348  used[_vindex(*vs.begin())] = next;// always cover by var node once visited
349  else
350  for (variable_set::const_iterator v = vs.begin(); v != vs.end();
351  ++v) {
352  if (used[_vindex(*v)] == vindex(-1))
353  used[_vindex(*v)] = next;// otherwise remember which factor we are
354  }
355  // if we have a parent, add an edge to our trees
356  if (nFound > 0)
357  tree.push_back(edge_t(parent, next)); //, _eindex(parent,next),_eindex(next,parent))); //!!!
358  // if this is a root, do nothing (!!!); don't keep track of isolated nodes?
359 
360  const set<edge_id> nbrs = neighbors(next);// add all neighbors to the search queue
361  for (set<edge_id>::const_iterator n = nbrs.begin(); n != nbrs.end();
362  ++n) {
363  Q->push(n->second);
364  }
365  }
366  delete Q;
367  return tree;
368  }
