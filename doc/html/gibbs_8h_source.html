132  void run() {
133 
134  double score = 0.0;
135  for (size_t f = 0; f < num_factors(); ++f)
136  score += std::log(get_factor(f)[sub2ind(get_factor(f).vars(), m_state)]);
137 
138  for (size_t j = 0, i = 0; i < m_num_samples; ++i) {// keep nSamples evenly space samples
139  size_t jNext = (size_t) ((1.0 + i) / m_num_samples * m_num_iter); // among nIter steps
140  for (; j < jNext; ++j) {
141 
142  // Each iteration, go over all the variables:
143  for (size_t v = 0; v < nvar(); ++v) {
144  if (var(v).states() == 0)
145  continue; // (make sure they're non-empty)
146  const flist& factors = with_variable(var(v)); // get the factors they are involved with
147  factor F(var(v), 1.0);
148  for (flist::const_iterator f = factors.begin();
149  f != factors.end(); ++f) {
150  variable_set vs = get_factor(*f).vars();
151  vs /= var(v); // and condition on their neighbors
152  F *= get_factor(*f).slice(vs, sub2ind(vs, m_state));
153  }
154  if (!m_no_best)
155  score -= std::log(F[m_state[v]]);// remove current value
156  if (m_temp != 1.0)
157  m_state[v] = (F ^ m_temp).sample(); // then draw a new value
158  else
159  m_state[v] = F.sample(); // (annealed or not)
160  if (!m_no_best) {
161  score += std::log(F[m_state[v]]);// update score incrementally
162  if (score > m_lb) {
163  m_lb = score;
164  m_best_config = m_state;
165  } // and keep the best
166  }
167  }
168 
169  // After each sweep, update our statistics
170  if (!m_no_best && isinf(score)) {// if we're still looking for a valid config
171  score = 0.0; // we need to update the score completely
172  for (size_t f = 0; f < num_factors(); ++f)
173  score += std::log(
174  get_factor(f)[sub2ind(get_factor(f).vars(), m_state)]);
175  }
176  if (!m_no_beliefs) { // if we're keeping marginal estimates
177  for (size_t f = 0; f < num_factors(); ++f) // then run through the factors
178  m_beliefs[f][sub2ind(get_factor(f).vars(), m_state)] += 1.0
179  / m_num_samples; // and update their status
180  }
181  if (m_temp_min != m_temp_max)
182  m_temp += (m_temp_max - m_temp_min) / m_num_iter;// update temperature if annealed
183 
184  }
185  m_samples.push_back(m_state);
186  }
187  }
