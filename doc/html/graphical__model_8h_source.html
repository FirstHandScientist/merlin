599  std::pair<size_t, size_t> pseudo_tree_size(const variable_order_t& order) const {
600  size_t width = 0, maxHeight = 0;
601  std::vector<size_t> heights;
602  heights.resize(nvar(), 0);
603  std::vector<variable_set> adj = mrf();
604 
605  // Eliminate in the given order of labels, tracking adjacency
606  for (variable_order_t::const_iterator i = order.begin(); i != order.end(); ++i) {
607  if (*i < 0 || *i >= nvar())
608  continue;
609  variable_set vi = adj[_vindex(var(*i))];
610  for (variable_set::const_iterator j = vi.begin(); j != vi.end(); ++j) {
611  adj[_vindex(*j)] |= vi;
612  adj[_vindex(*j)] -= variable_set(var(*i), *j);
613  heights[_vindex(*j)] = std::max(heights[_vindex(*j)],
614  heights[_vindex(var(*i))] + 1);
615  maxHeight = std::max(maxHeight, heights[_vindex(*j)]);
616  width = std::max(width, adj[_vindex(*j)].nvar());
617  }
618  }
619  return std::make_pair(width, maxHeight);
620  }

835  sMap scores;

837  double newScore;
838 
839  for (size_t v = 0; v < nvar(); v++) // get initial scores
840  reverse[v] = scores.insert(NN(order_score(adj, v, ord_type), v));
841 
842  for (size_t ii = 0; ii < nvar(); ++ii) {// Iterate through, selecting variables
843  sMap::iterator first = scores.begin();// Choose a random entry from among the smallest
844  newScore = std::max(first->first, newScore);// keep track of largest score for this ordering
845  if (newScore > old_score)
846  return; // & if worse than cutoff value, just quit
847  sMap::iterator last = scores.upper_bound(first->first);
848  std::advance(first, randi(std::distance(first, last)));
849  size_t i = first->second;
850

852  scores.erase(reverse[i]); // remove it from our list
853  variable_set vi = adj[i]; // go through adjacent variables (copy: adj may change)
854  variable_set fix; // and keep track of which need updating
855  for (variable_set::const_iterator j = vi.begin(); j != vi.end(); ++j) {
856  size_t v = _vindex(*j);
857  adj[v] |= vi; // and update their adjacency structures
858  adj[v] /= var(i);
859  if (ord_type == OrderMethod::MinWidth
860  || ord_type == OrderMethod::WtMinWidth)
861  fix |= adj[v]; //var(v); // (width methods only need v, not nbrs)
862  else
863  fix |= adj[v]; // come back and recalculate their scores
864  }
865  for (variable_set::const_iterator j = fix.begin(); j != fix.end(); ++j) {
866  size_t jj = j->label();
867  scores.erase(reverse[jj]); // remove and update (score,index) pairs
868  reverse[jj] = scores.insert(NN(order_score(adj, jj, ord_type), jj));
869  }
870  }

1024  double order_score(const std::vector<variable_set>& adj, size_t i, OrderMethod kOType) const {
1025  double s = 0.0;
1026  switch (kOType) {
1027  case OrderMethod::MinFill:
1028  for (variable_set::const_iterator j = adj[i].begin(); j != adj[i].end(); ++j)
1029  s += (adj[i] - adj[_vindex(*j)]).size();
1030  break;
1031  case OrderMethod::WtMinFill:
1032  for (variable_set::const_iterator j = adj[i].begin(); j != adj[i].end(); ++j)
1033  s += (adj[i] - adj[*j]).num_states();
1034  break;
1035  case OrderMethod::MinWidth:
1036  s = adj[i].size();
1037  break;
1038  case OrderMethod::WtMinWidth:
1039  s = adj[i].num_states();
1040  break;
1041  default:
1042  throw std::runtime_error("Unknown elimination ordering type");
1043  break;
1044  }
1045  return s;
1046  }
