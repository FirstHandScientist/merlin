38  subindex(const variable_set& full, const variable_set& sub) {
39  assert( full >> sub);
40  m_idx = 0;
41  m_end = 1;
42  m_nd = full.nvar();
43  m_dims = full.dims();
44  m_state = new vsize[m_nd];
45  m_add = new vsize[m_nd];
46  m_subtract = new vsize[m_nd];
47  m_skipped = new bool[m_nd];
48  // Compute reference index updates
49  vsize i, j;
50  for (i = 0, j = 0; i < m_nd; ++i) {
51  m_state[i] = 1; // start with [0] = (1,1,1,1...)
52  m_skipped[i] = (j >= sub.nvar() || sub[j] != full[i]); // are we sub-indexing this variable?
53  if (i == 0)
54  m_add[i] = 1; // how much does adding one to this var
55  else
56  m_add[i] = m_add[i - 1] * (m_skipped[i - 1] ? 1 : m_dims[i - 1]); // add to our position?
57  m_subtract[i] = m_add[i] * ((m_skipped[i] ? 1 : m_dims[i]) - 1); // how much does wrapping back to 1 remove from pos?
58  if (!m_skipped[i])
59  j++;
60  m_end *= m_dims[i];
61  }
62  //end_--;
63  }

195  superindex(const variable_set& full, const variable_set& sub, size_t offset) {
196  assert( full >> sub);
197  m_idx = offset;
198  m_end = full.num_states();
199  m_ns = sub.nvar();
200  const vsize* dimf = full.dims();
201  m_dims = sub.dims();
202  m_state = new vsize[m_ns];
203  m_add = new vsize[m_ns];
204  size_t i, j, d = 1;
205  for (i = 0, j = 0; j < m_ns; ++i) {
206  if (full[i] == sub[j]) {
207  m_state[j] = 1;
208  m_add[j] = d;
209  j++;
210  }
211  d *= dimf[i];
212  }
213  m_end = m_add[m_ns - 1] * m_dims[m_ns - 1] + offset;
214  }

300  permute_index(const std::vector<variable>& order, bool big_endian = false) {
301  m_i = 0;
302  variable_set _vs = variable_set(order.begin(), order.end(), order.size()); // compute implicit source order (VarSet)
303  m_pi.resize(order.size());
304  m_dim.resize(order.size());
305  for (size_t j = 0; j < order.size(); ++j)
306  m_dim[j] = _vs[j].states(); // save dimensions in source order (VarSet)
307  for (size_t j = 0; j < order.size(); ++j) { // compute mapping from target order to source
308  size_t jj = big_endian ? order.size() - 1 - j : j;
309  for (size_t k = 0; k < order.size(); ++k)
310  if (_vs[k] == order[j]) {
311  m_pi[jj] = k;
312  break;
313  }
314  }
315  }
