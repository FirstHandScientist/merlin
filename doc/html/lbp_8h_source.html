185  // Prologue

193  m_beliefs = std::vector<merlin::factor>(m_factors); // copy initial beliefs from factors
194  m_msg = std::vector<merlin::factor>();
195  m_msg.resize(2 * num_edges()); // initialize messages to the identity
196  for (size_t e = 0; e < 2 * num_edges(); ++e)
197  if (edge(e) != edge_id::NO_EDGE) { // f'n of the right variables
198  m_msg[e] = merlin::factor(
199  merlin::factor(edge(e).first).vars()
200  & factor(edge(e).second).vars(), 1.0);
201  }
202  m_msg_new = std::vector<merlin::factor>(m_msg); // copy that as "updated" message list
203 
204  m_log_z = 0.0; // compute initial partition f'n estimate
205  for (size_t f = 0; f < num_factors(); ++f) {
206  bel(f) /= bel(f).sum(); // normalize the beliefs
207  m_log_z += (bel(f) * log(factor(f))).sum() + obj_entropy(f); // and compute the free energy estimate
208  }
209 
210  if (m_sched == Schedule::Priority) { // for priority scheduling
211  for (size_t e = 0; e < 2 * num_edges(); ++e) // initialize all edges to infinity
212  if (edge(e) != edge_id::NO_EDGE)
213  m_priority.insert(std::numeric_limits<double>::infinity(), e);
214  } else {
215  for (size_t f = 0; f < num_factors(); ++f)
216  m_forder.push_back(f); // for fixed scheduling, get a default order
217  }

261  size_t stopIter = m_stop_iter * num_factors();// it's easier to count updates than "iterations"
262 
263  double dObj = m_stop_obj + 1.0, dMsg = m_stop_msg + 1.0;// initialize termination values
264  size_t iter = 0, print = 1; // count updates and "iterations" for printing
265  size_t f, n = 0; //
266 
267  std::cout << "Begin message passing over factor graph ..." << std::endl;
268  std::cout << "- stopObj : " << m_stop_obj << std::endl;
269  std::cout << "- stopMsg : " << m_stop_msg << std::endl;
270  std::cout << "- stopIter : " << m_stop_iter << std::endl;
271  for (; dMsg >= m_stop_msg && iter < stopIter && dObj >= m_stop_obj;) {
272 
273  if (m_sched == Schedule::Priority) { // priority schedule =>
274  f = edge(m_priority.top().second).second; // get next factor for update from queue
275  m_priority.pop();
276  } else { // fixed schedule =>
277  f = m_forder[n]; // get next factor from list
278  if (++n == m_forder.size())
279  n = 0;
280  }
281 
282  if (m_sched != Schedule::Flood) { // For non-"flood" schedules,
283  merlin::factor logF = log(factor(f));
284  dObj = 0.0; // compute new belief and update objective:
285  dObj -= (belief(f) * logF).sum() + obj_entropy(f); // remove old contribution
286  accept_incoming(f); // accept all messages into factor f
287  m_log_z += dObj += (belief(f) * logF).sum() + obj_entropy(f); // re-add new contribution
288  }
289  update_outgoing(f); // update outgoing messages from factor f
290 
291  if (m_sched == Schedule::Priority)
292  dMsg = m_priority.top().first; // priority schedule => easy to check msg stop
293  else if (m_stop_msg > 0 && n == 0) { // else check once each time through all factors
294  dMsg = 0.0;
295  for (size_t e = 0; e < 2 * num_edges(); ++e)
296  dMsg = std::max(dMsg, m_msg_new[e].distance(m_msg[e], m_dist));
297  }
298 
299  if (m_sched == Schedule::Flood && n == 0) { // for flooding schedules, recalculate all
300  dObj = m_log_z;
301  m_log_z = 0.0; // the beliefs and objective now
302  for (size_t f = 0; f < num_factors(); ++f) {
303  accept_incoming(f);
304  m_log_z += (belief(f) * log(factor(f))).sum() + obj_entropy(f);
305  }
306  dObj -= m_log_z;
307  }
308 
309  if (iter > print * num_factors()) {
310  print++;
311  // std::cout << " LBP: " << _lnZ << " \t\t(dObj=" << dObj << " \t dMsg=" << dMsg << ")\n";

323  }

391  void accept_incoming(size_t n) { //
392  const set<edge_id>& nbrs = neighbors(n); // get the list of neighbors
393  bel(n) = factor(n); // and start with just the local factor
394  for (set<edge_id>::const_iterator i = nbrs.begin(); i != nbrs.end();
395  ++i) {
396  m_msg[i->ridx] = m_msg_new[i->ridx]; // accept each new incoming message
397  bel(n) *= m_msg[i->ridx]; // and include it in the belief
398  if (m_sched == Schedule::Priority)
399  m_priority.erase(i->ridx); // accepted => remove from priority_ queue
400  }
401  bel(n) /= bel(n).sum(); // normalize belief
402  }

417  if (m_sched == Schedule::Priority) // and update m_priority in schedule
