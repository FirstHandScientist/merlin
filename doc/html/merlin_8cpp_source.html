113 bool Merlin::read_evidence(const char* f) {
114  try {
115 
116  // Open the evidence file
117  std::ifstream in(f);
118  if (in.fail()) {
119  throw std::runtime_error("Error while opening the evidence file.");
120  }
121 
122  // Clear any previous evidence
123  m_evidence.clear();
124 
125  // Read the evidence file
126  int num_evid;
127  in >> num_evid;
128  for (int i = 0; i < num_evid; ++i) {
129  vindex var;
130  size_t val;
131  in >> var >> val;
132  m_evidence[var] = val;
133  }
134 
135  // Close the evidence file
136  in.close();
137 
138  return true;
139  } catch (const std::runtime_error& e) {
140  std::cerr << e.what() << std::endl;
141  return false;
142  }

183 void Merlin::check() {
184  if (m_task == MERLIN_TASK_PR) {
185  if (m_algorithm != MERLIN_ALGO_WMB) {
186  throw std::runtime_error("Unsupported PR inference algorithm. Use WMB.");
187  }
188  } else if (m_task == MERLIN_TASK_MAR) {
189  if (m_algorithm != MERLIN_ALGO_WMB &&
190  m_algorithm != MERLIN_ALGO_IJGP &&
191  m_algorithm != MERLIN_ALGO_LBP &&
192  m_algorithm != MERLIN_ALGO_GIBBS) {
193  throw std::runtime_error("Unsupported MAR inference algorithm. Use WMB, IJGP, LBP, GIBBS.");
194  }
195  } else if (m_task == MERLIN_TASK_MAP) {
196  if (m_algorithm != MERLIN_ALGO_WMB &&
197  m_algorithm != MERLIN_ALGO_JGLP &&
198  m_algorithm != MERLIN_ALGO_IJGP &&
199  m_algorithm != MERLIN_ALGO_GIBBS) {
200  throw std::runtime_error("Unsupported MAP inference algorithm. Use WMB, JGLP, IJGP, GIBBS.");
201  }
202  } else if (m_task == MERLIN_TASK_MMAP) {
203  if (m_algorithm != MERLIN_ALGO_WMB) {
204  throw std::runtime_error("Unsupported MMAP inference algorithm. Use WMB.");
205  }
206  } else {
207  throw std::runtime_error("Unsupported inference task. Use PR, MAR, MAP, MMAP");
208  }
209 }

214 int Merlin::run() {
215 
216  try {
217 
218  // Safety checks
219  check();
220 
221  // Initialize the graphical model
222  merlin::graphical_model gm;
223  std::vector<merlin::factor> fs;
224  std::map<vindex, vindex> old2new;
225  gm = *(static_cast<merlin::graphical_model*>(m_gmo));
226 
227  // Assert the evidence
228  if ( m_evidence.empty() == false ) {
229  fs = gm.assert_evidence( m_evidence, old2new );
230  } else {
231  fs = gm.get_factors();
232  for (size_t v = 0; v < gm.nvar(); ++v) {
233  old2new[v] = v;
234  }
235  }
236 
237  // Setup the solver to run
238  if ( m_task == MERLIN_TASK_PR ) {
239  std::string output_file = m_filename + ".PR.out";
240  merlin::wmb s(fs);
241  std::ostringstream oss;
242  oss << "iBound=" << m_param_ibound << ","
243  << "Order=MinFill" << ","
244  << "Iter=" << m_param_iterations << ","
245  << "Task=PR";
246  s.set_properties(oss.str());
247  s.run();
248  s.write(output_file.c_str(), m_evidence, old2new, gm);
249  } else if ( m_task == MERLIN_TASK_MAR ) {
250  std::string output_file = m_filename + ".MAR.out";
251  if (m_algorithm == MERLIN_ALGO_WMB) {
252  merlin::wmb s(fs);
253  std::ostringstream oss;
254  oss << "iBound=" << m_param_ibound << ","
255  << "Order=MinFill" << ","
256  << "Iter=" << m_param_iterations << ","
257  << "Task=MAR";
258  s.set_properties(oss.str());
259  s.run();
260  s.write(output_file.c_str(), m_evidence, old2new, gm);
261  } else if (m_algorithm == MERLIN_ALGO_IJGP) {
262  merlin::ijgp s(fs);
263  std::ostringstream oss;
264  oss << "iBound=" << m_param_ibound << ","
265  << "Order=MinFill" << ","
266  << "Iter=" << m_param_iterations << ","
267  << "Task=MAR";
268  s.set_properties(oss.str());
269  s.run();
270  s.write(output_file.c_str(), m_evidence, old2new, gm);
271  } else if (m_algorithm == MERLIN_ALGO_LBP) {
272  merlin::lbp s(fs);
273  std::ostringstream oss;
274  oss << "Schedule=Priority,Distance=HPM,stopIter="
275  << m_param_iterations << ",stopObj=-1,stopMsg=-1";
276  s.set_properties(oss.str());
277  s.run();
278  s.write(output_file.c_str(), m_evidence, old2new, gm);
279  } else if (m_algorithm == MERLIN_ALGO_GIBBS) {
280  merlin::gibbs s(fs);
281  std::ostringstream oss;
282  oss << "TempMin=1.0,TempMax=1.0,Best=1,Beliefs=0" << ","
283  << "nIter=" << m_param_iterations << ","
284  << "nSamples=" << m_param_samples;
285  s.set_properties(oss.str());
286  s.run();
287  s.write(output_file.c_str(), m_evidence, old2new, gm);
288  }
289  } else if ( m_task == MERLIN_TASK_MAP ) {
290  std::string output_file = m_filename + ".MAP.out";
291  if (m_algorithm == MERLIN_ALGO_WMB) {
292  merlin::wmb s(fs);
293  std::ostringstream oss;
294  oss << "iBound=" << m_param_ibound << ","
295  << "Order=MinFill" << ","
296  << "Iter=" << m_param_iterations << ","
297  << "Task=MAP";
298  s.set_properties(oss.str());
299  std::vector<vindex> qvars;
300  for (size_t i = 0; i < gm.nvar(); ++i) {
301  if (m_evidence.find(i) == m_evidence.end()) {
302  size_t nvar = old2new.at(i);
303  qvars.push_back(nvar); // use the new index of the MAP vars
304  }
305  }
306  s.set_query(qvars);
307  s.run();
308  s.write(output_file.c_str(), m_evidence, old2new, gm);
309  } else if (m_algorithm == MERLIN_ALGO_JGLP) {
310  merlin::jglp s(fs);
311  std::ostringstream oss;
312  oss << "iBound=" << m_param_ibound << ","
313  << "Order=MinFill" << ","
314  << "Iter=" << m_param_iterations;
315  s.set_properties(oss.str());
316  s.run();
317  s.write(output_file.c_str(), m_evidence, old2new, gm);
318  } else if (m_algorithm == MERLIN_ALGO_IJGP) {
319  merlin::ijgp s(fs);
320  std::ostringstream oss;
321  oss << "iBound=" << m_param_ibound << ","
322  << "Order=MinFill" << ","
323  << "Iter=" << m_param_iterations << ","
324  << "Task=MAP";
325  s.set_properties(oss.str());
326  s.run();
327  s.write(output_file.c_str(), m_evidence, old2new, gm);
328  } else if (m_algorithm == MERLIN_ALGO_GIBBS) {
329  merlin::gibbs s(fs);
330  std::ostringstream oss;
331  oss << "TempMin=1.0,TempMax=1.0,Best=0,Beliefs=1" << ","
332  << "nIter=" << m_param_iterations << ","
333  << "nSamples=" << m_param_samples;
334  s.set_properties(oss.str());
335  s.run();
336  s.write(output_file.c_str(), m_evidence, old2new, gm);
337  } // follow with search-based AOBB, AOBF, RBFAOO
338  } else if ( m_task == MERLIN_TASK_MMAP ) {
339  std::string output_file = m_filename + ".MMAP.out";
340  if (m_algorithm == MERLIN_ALGO_WMB) {
341  merlin::wmb s(fs);
342  std::ostringstream oss;
343  oss << "iBound=" << m_param_ibound << ","
344  << "Order=MinFill" << ","
345  << "Iter=" << m_param_iterations << ","
346  << "Task=MMAP";
347  s.set_properties(oss.str());
348  std::vector<size_t> qvars;
349  for (size_t i = 0; i < m_query.size(); ++i) {
350  vindex var = m_query[i];
351  vindex nvar = old2new.at(var);
352  qvars.push_back(nvar); // use the new index of the MAP vars
353  }
354  s.set_query(qvars);
355  s.run();
356  s.write(output_file.c_str(), m_evidence, old2new, gm);
357  } // follow with search-based AOBB, AOBF, RBFAOO
358  }
359 
360  return EXIT_SUCCESS;
361  } catch (const std::runtime_error& e) {
362  std::cerr << e.what() << std::endl;
363  return EXIT_FAILURE;
364  }
