422  double score(const vector<variable_set>& fin, const variable& VX, size_t i, size_t j) {
423  double err;
424  const variable_set& F1 = fin[i], &F2 = fin[j]; // (useful shorthand)
425  size_t iBound = std::max(std::max(m_ibound, F1.nvar() - 1),
426  F2.nvar() - 1); // always OK to keep same size
427  variable_set both = F1 + F2;
428  if (both.nvar() > iBound+1)
429  err = -3; // too large => -3

431  err = 1.0 / (F1.nvar() + F2.nvar()); // greedy scope-based 2 (check if useful???)
432  //else if (_byScope) err = 1; // scope-based => constant score
433  return err;
